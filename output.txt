declare i32 @printf(i8*, ...)
@print.str = constant [4 x i8] c"%d\0A\00"

define i32 @main() {

	%a = alloca i32
	%b = alloca i32
	% = alloca i32

	store i32 0, i32* %a
	store i32 0, i32* %b
	store i32 0, i32* %

	%t1 = add i32 2, 2
	%t2 = load i32* %b
	%t3 = mul i32 %t2, %t1
	%t4 = sub i32 %t3, 4
	store i32 %t4, i32* %a
	ret i32 0
}
